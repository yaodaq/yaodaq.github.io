44Identifier Identifier::parse( const std::string& id )
45{
46 std::vector<std::string> result;
47 std::string tmp = id;
48 std::string separator = "/";
49 std::size_t second_pos = tmp.find( separator );
50 while( second_pos != std::string::npos )
51 {
52 if( 0 != second_pos )
53 {
54 std::string word = tmp.substr( 0, second_pos - 0 );
55 result.push_back( word );
56 }
57 else
58 result.push_back( "" );
59 tmp = tmp.substr( second_pos + separator.length() );
60 second_pos = tmp.find( separator );
61 if( second_pos == std::string::npos ) result.push_back( tmp );
62 }
63 if( result.size() == 5 )
64 {
65 Identifier identifier( result[3], result[4] );
66 identifier.generateKey( magic_enum::enum_cast<Domain>( result[0] ).value(), magic_enum::enum_cast<Class>( result[1] ).value(), magic_enum::enum_cast<Family>( result[2] ).value() );
67 return identifier;
68 }
69 else
70 {
71 throw Exception( StatusCode::WRONG_NUMBER_PARAMETERS, "Number of parameters in key should be 5 (Domain/Class/Family/Type/Name) !" );
72 }
