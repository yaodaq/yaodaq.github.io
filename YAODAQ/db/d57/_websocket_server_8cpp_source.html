38 setOnClientMessageCallback(
39 [this]( std::shared_ptr<ix::ConnectionState> connectionState, ix::WebSocket& webSocket, const ix::WebSocketMessagePtr& msg )
40 {
41 // The ConnectionState object contains information about the connection
42 std::shared_ptr<ConnectionState> connection = std::static_pointer_cast<ConnectionState>( connectionState );
43 if( msg->type == ix::WebSocketMessageType::Message ) {}
44 else if( msg->type == ix::WebSocketMessageType::Open )
45 {
46 // Check if a client with the same name is already connected;
47 connection->computeId( getHost() + ":" + std::to_string( getPort() ), Identifier::parse( msg->openInfo.headers["id"] ) );
48 if( connection->isTerminated() )
49 {
50 logger()->error( fmt::format( fg( fmt::color::red ) | fmt::emphasis::bold, "One client with the name \"{}\" is already connected !", Identifier::parse( msg->openInfo.headers["id"] ).getName() ) );
51 webSocket.stop( magic_enum::enum_integer( StatusCode::CLIENT_WITH_SAME_NAME_ALREADY_CONNECTED ),
52 fmt::format( "One client with the name \"{}\" is already connected to ws{}://{}:{} !", Identifier::parse( msg->openInfo.headers["id"] ).getName(), "", getHost(), getPort() ) );
53 std::this_thread::sleep_for( std::chrono::milliseconds( 10 ) );
54 return;
55 }
56 addClient( Identifier::parse( msg->openInfo.headers["id"] ), webSocket );
57 Open open( msg->openInfo, connection );
58 sendToLoggers( open, webSocket );
59 onOpen( open );
60 }
61 else if( msg->type == ix::WebSocketMessageType::Close )
62 {
63 Close close( msg->closeInfo, connection );
64 sendToLoggers( close, webSocket );
65 onClose( close );
66 removeClient( webSocket );
67 }
68 else if( msg->type == ix::WebSocketMessageType::Error )
69 {
70 Error error( msg->errorInfo, connection );
71 sendToLoggers( error, webSocket );
72 onError( error );
73 }
74 else if( msg->type == ix::WebSocketMessageType::Ping )
75 {
76 Ping ping( msg, connection );

219 else if( value >= magic_enum::enum_integer( yaodaq::Severity::Error ) )

221 logger()->error( "Signal SIG{} raised !", magic_enum::enum_name( signal ) );

224 {
225 fmt::print( "\n" );
226 logger()->warn( "Signal SIG{} raised !", magic_enum::enum_name( signal ) );
227 }
228 else if( value >= magic_enum::enum_integer( yaodaq::Severity::Info ) )
229 {
230 fmt::print( "\n" );
231 logger()->info( "Signal SIG{} raised !", magic_enum::enum_name( signal ) );
232 }
233 else
234 {
235 fmt::print( "\n" );
236 logger()->trace( "Signal SIG{} raised !", magic_enum::enum_name( signal ) );
