77void WebsocketClient::onMessage( Message& message )
78{
79 switch( message.getTypeValue() )
80 {
81 case MessageType::Exception:
82 MessageException& message_exception = reinterpret_cast<MessageException&>( message );
83 // Special case for connection to the server with the same name as an other client !
84 if( static_cast<StatusCode>( message_exception.getCode() ) == StatusCode::CLIENT_WITH_SAME_NAME_ALREADY_CONNECTED )
85 {
86 disableAutomaticReconnection();
87 std::this_thread::sleep_for( std::chrono::milliseconds( 100 ) );
88 close( message_exception.getCode(), message_exception.getDescription() );
89 std::this_thread::sleep_for( std::chrono::milliseconds( 100 ) );
90 if( m_ThrowGeneralIfSameName ) throw Exception( static_cast<StatusCode>( message_exception.getCode() ), message_exception.getDescription() );
91 }
92 onException( message_exception );
93 break;
94 }
95}

155void WebsocketClient::onRaisingSignal()
156{
157 Signal signal = m_Looper.loop();
158 if( m_Looper.getInstance() == 0 )

160 int value = magic_enum::enum_integer( signal );
161 if( value >= magic_enum::enum_integer( yaodaq::Severity::Critical ) ) { logger()->critical( "Signal SIG{} raised !", magic_enum::enum_name( signal ) ); }
162 else if( value >= magic_enum::enum_integer( yaodaq::Severity::Error ) )
163 {
164 logger()->error( "Signal SIG{} raised !", magic_enum::enum_name( signal ) );
165 }
166 else if( value >= magic_enum::enum_integer( yaodaq::Severity::Warning ) )
167 {
168 fmt::print( "\n" );
169 logger()->warn( "Signal SIG{} raised !", magic_enum::enum_name( signal ) );
170 }
171 else if( value >= magic_enum::enum_integer( yaodaq::Severity::Info ) )
172 {
173 fmt::print( "\n" );
174 logger()->info( "Signal SIG{} raised !", magic_enum::enum_name( signal ) );
175 }
176 else
177 {
178 fmt::print( "\n" );
179 logger()->trace( "Signal SIG{} raised !", magic_enum::enum_name( signal ) );
180 }
181 if( magic_enum::enum_integer( signal ) >= magic_enum::enum_integer( Severity::Critical ) ) std::exit( magic_enum::enum_integer( signal ) );
